# ========================================== 챗봇 기능 ====================
# 역할: 사용자와 AI가 대화할 수 있는 챗봇 페이지
# 위치: /chatbot/ 주소로 접속

import json  # JSON 데이터를 읽고 쓰기 위한 라이브러리
from django.views.decorators.csrf import csrf_exempt  # API 보안 설정 (CSRF 토큰 제외)

def chatbot(request):
    """
    챗봇 페이지 보여주기
    - 사용자와 AI가 대화할 수 있는 채팅 인터페이스
    - 주소: /chatbot/
    """
    return render(request, 'pages/chatbot.html')


@csrf_exempt  # API 요청이므로 CSRF 토큰 검증 제외 (보안상 실제 운영에서는 주의 필요)
def chatbot_api(request):
    """
    챗봇 AI 응답 API
    - 사용자 메시지를 받아서 AI 응답 생성
    - POST /pages/chatbot/api/ {"message": "안녕하세요"}
    
    동작 순서:
    1. 사용자가 메시지 전송 (예: "안녕하세요")
    2. 서버가 메시지 받음
    3. generate_ai_response() 함수로 AI 응답 생성
    4. JSON 형식으로 응답 반환 (예: {"response": "안녕하세요! 무엇을 도와드릴까요?"})
    """
    # POST 요청이 아니면 에러
    if request.method != 'POST':
        return JsonResponse({'error': '잘못된 요청입니다'}, status=400)
    
    try:
        # 요청에서 사용자 메시지 받기
        data = json.loads(request.body)  # JSON 문자열을 Python 딕셔너리로 변환
        user_message = data.get('message', '').strip()  # 'message' 키의 값 가져오기
        
        # 메시지가 비어있으면 에러
        if not user_message:
            return JsonResponse({'error': '메시지를 입력해주세요'}, status=400)
        
        # 🤖 AI 응답 생성 (현재는 간단한 규칙 기반, 나중에 OpenAI API로 교체 가능)
        ai_response = generate_ai_response(user_message)
        
        # JSON 형식으로 응답 반환
        return JsonResponse({
            'success': True,
            'response': ai_response,  # AI가 생성한 답변
            'timestamp': timezone.now().isoformat()  # 현재 시간
        })
    
    except json.JSONDecodeError:
        # JSON 파싱 오류
        return JsonResponse({'error': 'JSON 형식이 올바르지 않습니다'}, status=400)
    except Exception as e:
        # 기타 오류
        return JsonResponse({'error': str(e)}, status=500)


def generate_ai_response(user_message):
    """
    사용자 메시지에 대한 AI 응답 생성
    - 현재는 간단한 규칙 기반 (키워드 매칭)
    - 실제 AI 연동 시 OpenAI API 또는 Google Gemini API 사용
    
    동작 원리:
    1. 사용자 메시지를 소문자로 변환
    2. 특정 키워드가 있는지 확인 (예: "안녕", "기능", "신고" 등)
    3. 키워드에 맞는 답변 반환
    4. 매칭되는 키워드가 없으면 기본 응답
    """
    message_lower = user_message.lower()  # 소문자로 변환 (대소문자 구분 없이 비교)
    
    # 인사말 체크
    if any(word in message_lower for word in ['안녕', 'hi', 'hello', '헬로', '하이']):
        return "안녕하세요! 😊 SafeMap 챗봇입니다. 무엇을 도와드릴까요?"
    
    # 기능 문의
    elif any(word in message_lower for word in ['기능', '뭐', '무엇', '할 수 있어', 'what']):
        return """저는 SafeMap 챗봇입니다! 다음과 같은 도움을 드릴 수 있어요:

📍 안전 신고 방법 안내
🗺️ 지역별 사건사고 정보 조회
📝 게시판 사용법 안내
❓ 자주 묻는 질문 답변

궁금하신 내용을 말씀해주세요!"""
    
    # 신고 관련
    elif any(word in message_lower for word in ['신고', '제보', 'report', '사건', '사고']):
        return """🚨 안전 신고는 다음과 같이 할 수 있습니다:

1. 상단 메뉴에서 '신고하기' 클릭
2. 지도에서 사건 발생 위치 선택
3. 카테고리와 위험도 선택
4. 상세 내용 작성 후 제출

💡 Tip: 사진을 첨부하면 더욱 정확한 신고가 가능해요!"""
    
    # 게시판 관련
    elif any(word in message_lower for word in ['게시판', '글', '작성', 'board', '게시글']):
        return """📝 게시판 사용법:

• 글 보기: 메뉴 → 게시판
• 글 쓰기: 게시판 우측 하단 '글 작성' 버튼
• 로그인 필수: 게시글 작성/수정/삭제는 로그인 후 가능합니다

본인이 작성한 글만 수정/삭제할 수 있어요!"""
    
    # 지도 관련
    elif any(word in message_lower for word in ['지도', 'map', '위치', '지역']):
        return """🗺️ 지도 사용법:

• 시/도 선택 → 시/군/구 선택 → 동 선택
• 날짜 필터로 최근 7일/30일/90일 조회
• 카테고리별 필터링 가능
• 위험도 범위 설정 가능
• 상단 미니 리스트에서 최근 제보 확인

지도를 클릭하면 해당 지역 상세 정보를 볼 수 있어요!"""
    
    # 로그인 관련
    elif any(word in message_lower for word in ['로그인', 'login', '회원가입', '계정', 'sign']):
        return """🔐 로그인/회원가입:

• 회원가입: 우측 상단 'Sign Up' 클릭
• 로그인: 우측 상단 'Login' 클릭
• 로그인 후 가능한 기능:
  - 게시글 작성/수정/삭제
  - 안전 신고 제보
  - 내 활동 내역 조회

계정이 없으시면 먼저 회원가입을 진행해주세요!"""
    
    # 도움말
    elif any(word in message_lower for word in ['도움', 'help', '헬프', '사용법']):
        return """❓ SafeMap 사용 가이드:

🏠 홈: 최근 통계 및 대시보드
🗺️ 지도: 지역별 사건사고 조회
📝 게시판: 커뮤니티 게시판
📍 신고: 안전 제보 작성
💬 챗봇: 실시간 도움말 (지금 여기!)

각 메뉴를 클릭해서 기능을 사용해보세요!"""
    
    # 감사 인사
    elif any(word in message_lower for word in ['고마워', '감사', 'thank', '땡큐']):
        return "천만에요! 😊 더 궁금하신 점이 있으면 언제든 물어보세요!"
    
    # 작별 인사
    elif any(word in message_lower for word in ['잘가', '바이', 'bye', '안녕']):
        return "안녕히 가세요! 👋 SafeMap을 이용해주셔서 감사합니다. 안전한 하루 되세요!"
    
    # 기본 응답 (매칭되는 키워드가 없을 때)
    else:
        return f"'{user_message}'에 대해 잘 모르겠어요. 😅\n\n다음과 같이 질문해보세요:\n• '기능이 뭐야?'\n• '신고는 어떻게 해?'\n• '게시판 사용법'\n• '지도 사용법'"

from . import views

app_name = 'pages'
urlpatterns = [
    path('', views.home, name='home'),

    path('api/kr-admin/', views.kr_admin, name='kr_admin'),
    
    # 게시판 ==========================================혜은
    path('board/', views.board_list, name='board_list'),
    path('board/<int:pk>/', views.board_detail, name='board_detail'),

    # 게시글 작성 ==========================================혜은

    #    JavaScript에서 POST 요청으로 사용

]

// 🚀 챗봇 JavaScript - 채팅 기능 구현

// HTML 요소 가져오기
const chatMessages = document.getElementById('chatMessages');  // 채팅 메시지 영역
const chatInput = document.getElementById('chatInput');  // 입력창
const sendButton = document.getElementById('sendButton');  // 전송 버튼

// Enter 키로도 전송 가능
chatInput.addEventListener('keypress', function(e) {
    if (e.key === 'Enter' && !e.shiftKey) {  // Enter 키 (Shift 키 안 눌림)
        e.preventDefault();  // 기본 동작 취소 (줄바꿈 방지)
        sendMessage();  // 메시지 전송
    }
});

// 빠른 질문 버튼 클릭
function sendQuickQuestion(question) {
    chatInput.value = question;  // 입력창에 질문 넣기
    sendMessage();  // 메시지 전송
}

// 메시지 전송 함수 (핵심 기능!)
async function sendMessage() {
    const message = chatInput.value.trim();  // 입력값 가져오기 (공백 제거)
    
    // 빈 메시지는 전송 안 함
    if (!message) {
        return;
    }
    
    // 환영 메시지 숨기기
    const welcomeMsg = chatMessages.querySelector('.welcome-message');
    if (welcomeMsg) {
        welcomeMsg.style.display = 'none';
    }
    
    // 사용자 메시지 화면에 추가
    addMessage(message, 'user');
    
    // 입력창 비우기
    chatInput.value = '';
    
    // 버튼 비활성화 (중복 전송 방지)
    sendButton.disabled = true;
    sendButton.textContent = '전송 중...';
    
    // 타이핑 인디케이터 표시 (AI가 답변 중...)
    showTypingIndicator();
    
    try {
        // 서버에 메시지 전송 (AJAX 요청)
        const response = await fetch('{% url "pages:chatbot_api" %}', {
            method: 'POST',  // POST 방식
            headers: {
                'Content-Type': 'application/json',  // JSON 형식
            },
            body: JSON.stringify({ message: message })  // 메시지를 JSON으로 변환
        });
        
        const data = await response.json();  // 응답을 JSON으로 파싱
        
        // 타이핑 인디케이터 숨기기
        hideTypingIndicator();
        
        if (data.success) {
            // AI 응답을 화면에 추가
            addMessage(data.response, 'bot');
        } else {
            // 에러 메시지
            addMessage('죄송합니다. 오류가 발생했습니다: ' + (data.error || '알 수 없는 오류'), 'bot');
        }
    } catch (error) {
        hideTypingIndicator();
        addMessage('죄송합니다. 서버와 통신 중 오류가 발생했습니다. 😥', 'bot');
        console.error('Error:', error);  // 콘솔에 오류 출력
    } finally {
        // 버튼 다시 활성화
        sendButton.disabled = false;
        sendButton.textContent = '전송';
        
        // 입력창에 포커스
        chatInput.focus();
    }
}

// 메시지를 화면에 추가하는 함수
function addMessage(text, type) {
    const messageDiv = document.createElement('div');  // div 요소 생성
    messageDiv.className = `chat-message ${type}`;  // 클래스 추가 (user 또는 bot)
    
    const avatar = document.createElement('div');
    avatar.className = 'message-avatar';
    avatar.textContent = type === 'user' ? '👤' : '🤖';  // 아바타 아이콘
    
    const content = document.createElement('div');
    content.className = 'message-content';
    content.textContent = text;  // 메시지 내용
    
    // 사용자 메시지: 메시지 + 아바타 순서
    // AI 메시지: 아바타 + 메시지 순서
    if (type === 'user') {
        messageDiv.appendChild(content);
        messageDiv.appendChild(avatar);
    } else {
        messageDiv.appendChild(avatar);
        messageDiv.appendChild(content);
    }
    
    chatMessages.appendChild(messageDiv);  // 채팅 영역에 추가
    
    // 스크롤을 최신 메시지로 이동
    chatMessages.scrollTop = chatMessages.scrollHeight;
}

// 타이핑 인디케이터 표시 (AI가 답변 중...)
function showTypingIndicator() {
    const indicator = document.createElement('div');
    indicator.className = 'chat-message bot';
    indicator.id = 'typingIndicator';
    
    const avatar = document.createElement('div');
    avatar.className = 'message-avatar';
    avatar.textContent = '🤖';
    
    const typing = document.createElement('div');
    typing.className = 'typing-indicator';
    typing.style.display = 'block';
    typing.innerHTML = '<span></span><span></span><span></span>';  // 점 3개
    
    indicator.appendChild(avatar);
    indicator.appendChild(typing);
    chatMessages.appendChild(indicator);
    
    chatMessages.scrollTop = chatMessages.scrollHeight;
}

// 타이핑 인디케이터 숨기기
function hideTypingIndicator() {
    const indicator = document.getElementById('typingIndicator');
    if (indicator) {
        indicator.remove();  // 제거
    }
}

// 페이지 로드 시 입력창에 포커스
window.addEventListener('load', () => {
    chatInput.focus();
});

================================================================================
❓ [9단계] 자주 묻는 질문 (FAQ)
================================================================================

Q1. 챗봇이 응답을 안 해요
A1. 다음을 확인하세요:
    - 서버가 실행 중인지 (python manage.py runserver)
    - 브라우저 콘솔에 오류가 있는지 (F12 → Console 탭)
    - 메시지가 빈 문자열이 아닌지

Q2. "CSRF token missing" 오류가 나요
A2. views.py의 chatbot_api 함수에 @csrf_exempt 데코레이터가 있는지 확인

Q3. 챗봇이 멍청해요 (키워드만 인식)
A3. 현재는 규칙 기반 챗봇입니다. OpenAI API를 연동하면 더 똑똑해집니다.

Q4. 메뉴에 챗봇 링크가 안 보여요
A4. base.html을 수정했는지 확인하세요.

Q5. 채팅 기록이 저장되나요?
A5. 아니요, 새로고침하면 초기화됩니다. 
    저장하려면 데이터베이스 모델을 만들어야 합니다.
